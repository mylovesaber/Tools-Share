# 部署
## 同步测试

```bash
# 部署同步测试，需要运行一次查看首次添加信息，一次加上 -y 确认添加，再来一次查看重复情况，目前部署功能没有对 cron 定时进行严格的合法性检查（-O 和 -E），所以可以随便设置一个防止真正运行以检查预期表现。
# 文件夹
bash <(cat multi-sync-backup.sh) --sync_source_path "/root/tsource/" --sync_dest_path "/root/tdest" --sync_source_alias "ax108" --sync_dest_alias "ax119" --days 8 -G "test" -T "dir" -D "aaaa-aa-aa" -N "sync108-119" -O "12" -o "108-119" -l "test" -E "12" -L "ax108"

# 文件
bash <(cat multi-sync-backup.sh) --sync_source_path "/root/ts/" --sync_dest_path "/root/td" --sync_source_alias "ax108" --sync_dest_alias "ax119" --days 8 -G "test" -T "file" -D "aaaa-aa-aa" -N "sync108-119" -O "12" -o "108119" -l "test" -E "12" -L "ax108"

```
